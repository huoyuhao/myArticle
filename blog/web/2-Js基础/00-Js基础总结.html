# 1. 去除字符串两边空格

const trim = (str, type) => {
  const regObj = {
    left: /^\s+/,
    right: /\s+$/,
    both: /(^\s+)|(\s+$)/g,
    all: /\s+/g
  };
  const reg = (type && regObj[type]) ?? regObj.both;
  return str.replace(reg, "");
};

# 3. 字符串大小写切换

const caseConvert = (str) => {
  return str.replace(/([a-z]*)([A-Z]*)/g, (m, s1, s2) => {
    return `${s1.toUpperCase()}${s2.toLowerCase()}`;
  });
};
console.log(caseConvert('AbCdE')); // aBcDe
