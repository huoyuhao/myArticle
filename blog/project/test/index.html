["ab1","c1","ab2","d","c2"]
要求：
1. 未重复出现的字符串不处理；
2. 仅对相同的一组字符串依次添加序号，而不是共用一组序号；
3. 保持原数组顺序；
*/
function addSerial(arr) {
  var result = []
  // do something
  return result
}
console.log(addSerial(["ab", "c", "ab", "d", "c"]));

function addSerial(arr) {
  var result = []
  // do something
  let len = arr.length
  let obj = {}
  for(let i = 0; i < len; i++) {
    if (obj[arr[i]]) {
      obj[arr[i]].number++;
      arr[i] = arr[i] + obj[arr[i]].number;
    } else {
      obj[arr[i]] = {};
      obj[arr[i]].number = 1;
      obj[arr[i]].index = i;
    }
  }
  for(let key in obj){
    if (obj[key].number > 1) {
      arr[obj[key].index] = arr[obj[key].index] + '1';
    }
  }
  result = arr
  return result
}
console.log(addSerial(["ab", "c", "ab", "d", "c"]))
